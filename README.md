# tp-cloud


## to run : 

    cd app
    docker build -t tp-cloud .
    docker run -it -p 9000:3000 -v $(pwd):/app tp-cloud
    #or
    docker run -d -p 9000:3000 -v $(pwd):/app tp-cloud  
    #if you want to run it on background
